if (!book) { titleElement.innerText = "Kitob topilmadi"; return; } titleElement.innerText = book.title; bannerMessage.innerText = book.banner; // Mavzularni ekranga chiqarish container.innerHTML = ""; book.units.forEach(unit => { const card = document.createElement("a"); card.className = "chapter-card"; card.href = `training.html?book=${currentBookCode}&unit=${unit.id}`; card.innerHTML = `

${unit.name}
